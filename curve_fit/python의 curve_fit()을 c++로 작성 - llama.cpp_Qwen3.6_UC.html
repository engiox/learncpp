    int iter = 0;
    bool success = false;

    // 잔차 계산
    auto compute_residuals = [&](const std::vector<double>& par, std::vector<double>& res) {
        std::vector<double> f(n);
        model(x, par, f);
        for (size_t i = 0; i < n; ++i) res[i] = y[i] - f[i];
    };

    // 수치적 야코비안 계산
    auto compute_jacobian = [&](const std::vector<double>& par, Eigen::MatrixXd& J) {
        J.resize(n, m);
        std::vector<double> f0(n);
        model(x, par, f0);
        std::vector<double> p_plus = par;
        for (size_t j = 0; j < m; ++j) {
            p_plus[j] += eps;
            std::vector<double> f1(n);
            model(x, p_plus, f1);
            for (size_t i = 0; i < n; ++i) {
                J(i, j) = (f1[i] - f0[i]) / eps;
            }
            p_plus[j] -= eps;
        }
    };

    Eigen::VectorXd r(n);
    Eigen::MatrixXd J(n, m);
    double sse;
    std::vector<double> res(n);

    compute_residuals(p, res);
    sse = 0;
    for (double val : res) sse += val * val;

    for (iter = 0; iter < max_iter; ++iter) {
        compute_jacobian(p, J);

        compute_residuals(p, res);
        r = Eigen::Map<Eigen::VectorXd>(res.data(), n);

        // (J^T J + λ diag(J^T J)) dp = -J^T r
        Eigen::MatrixXd JtJ = J.transpose() * J;
        Eigen::MatrixXd A = JtJ + lam * JtJ.diagonal().asDiagonal();
        Eigen::VectorXd Jtr = J.transpose() * r;

        Eigen::VectorXd dp = A.colPivHouseholderQr().solve(-Jtr);

        // 파라미터 변화량 수렴 판정
        double max_dp = 0;
        for (size_t j = 0; j < m; ++j) max_dp = std::max(max_dp, std::abs(dp[j]));
        if (max_dp < tol) { success = true; break; }

        std::vector<double> p_new = p;
        for (size_t j = 0; j < m; ++j) p_new[j] += dp[j];

        // 경계(Bounds) 적용
        if (!lower.empty() && !upper.empty()) {
            bool in_bounds = true;
            for (size_t j = 0; j < m; ++j) {
                if (p_new[j] < lower[j] || p_new[j] > upper[j]) {
                    in_bounds = false;
                    break;
                }
            }
            if (!in_bounds) {
                lam *= 10.0;
                continue;
            }
        }

        // 단계 시도
        std::vector<double> res_new(n);
        compute_residuals(p_new, res_new);
        double sse_new = 0;
        for (double val : res_new) sse_new += val * val;

        if (sse_new < sse) {
            p = p_new;
            sse = sse_new;
            lam = std::max(lam * 0.1, 1e-10); // 성공 시 λ 감소
        } else {
            lam *= 10.0; // 실패 시 λ 증가
            if (lam > 1e10) break;
        }
    }

    // 최종 야코비안으로 공분산 행렬 계산
    compute_jacobian(p, J);
    compute_residuals(p, res);
    sse = 0;
    for (double val : res) sse += val * val;

    Eigen::MatrixXd cov;
    try {
        cov = (J.transpose() * J).ldlt().solve(Eigen::MatrixXd::Identity(m, m));
        // degrees of freedom로 스케일링
        cov *= sse / std::max(1.0, static_cast<double>(n - m));
    } catch (...) {
        cov = Eigen::MatrixXd::Zero(m, m); // 특이행렬 또는 수렴 실패 시 0 행렬
    }

    return {p, cov, sse, success, iter};
}

2. 사용 예시 (Python과 유사한 형태)

cpp
#include "curve_fit.hpp"

int main() {
    // 1. 데이터 생성 (예: 지수 감쇠 모델)
    std::vector<double> x = {0, 1, 2, 3, 4, 5};
    std::vector<double> y = {1.0, 0.82, 0.55, 0.38, 0.25, 0.18};
    std::vector<double> noise = {0.03, -0.02, 0.04, -0.01, 0.02, -0.03};
    for (size_t i = 0; i < x.size(); ++i) y[i] += noise[i];

    // 2. 모델 정의: y = p[0] * exp(-p[1] * x) + p[2]
    auto model = [](const std::vector<double>& x, const std::vector<double>& p, std::vector<double>& out) {
        out.resize(x.size());
        for (size_t i = 0; i < x.size(); ++i) {
            out[i] = p[0] * std::exp(-p[1] * x[i]) + p[2];
        }
    };

    // 3. 초기 파라미터 설정
    std::vector<double> p0 = {0.9, 0.5, 0.1};
    // 옵션: bounds = {lower, upper}
    std::vector<double> lower = {0.0, 0.1, 0.0};
    std::vector<double> upper = {2.0, 2.0, 0.5};

    // 4. 피팅 수행
    auto result = curve_fit(x, y, model, p0, lower, upper);

    // 5. 결과 출력
    std::cout << "✅ 최적 파라미터: ";
    for (size_t i = 0; i < result.params.size(); ++i) {
        std::cout << "p" << i << "=" << result.params[i];
        if (i < result.params.size()-1) std::cout << ", ";
    }
    std::cout << "\n";
    std::cout << "📉 Chi2 (SSE): " << result.chi2 << "\n";
    std::cout << "🎯 Success: " << (result.success ? "Yes" : "No") << "\n";
    std::cout << "🔄 Iterations: " << result.iterations << "\n";
    
    std::cout << "\n📊 공분산 행렬:\n" << result.covariance << "\n";
    std::cout << "📏 표준편차(σ): ";
    for (int i = 0; i < result.covariance.rows(); ++i) {
        std::cout << "σ" << i << "=" << std::sqrt(result.covariance(i, i));
        if (i < result.covariance.rows()-1) std::cout << ", ";
    }
    std::cout << "\n";

    return 0;
}

3. Python curve_fit과의 주요 차이점 및 참고사항

항목 Python (scipy.optimize.curve_fit) 위 C++ 구현
알고리즘 Levenberg-Marquardt (MINPACK 기반) 동일 (LM 알고리즘 직접 구현)
야코비안 자동으로 수치 미분 또는 사용자 정의 가능 기본적으로 수치 미분(Forward Difference) 사용
공분산 스케일링 pcov = (J^T J)^-1 * chi2 / (n-m) 동일
Bounds bounds=(lower, upper) 지원 배열 전달 방식 지원 (단계 거부 방식)
수렴 기준 xtol, ftol, gtol 조합 max_dp < tol (파라미터 변화량 기준)
컴파일 의존성 numpy, scipy Eigen3 (헤더 전용 라이브러리)

4. 성능 향상 팁 (프로덕션용)

  1. 분석적 야코비안 제공: 수치 미분은 O(n*m) 비용이 발생합니다. 모델이 미분 가능하다면 야코비안을 직접 계산하는 콜백을 추가하면 속도가 수십 배 빨라집니다.
  2. lmfit-cpp 또는 minpack 사용: Fortran MINPACK의 C++ 포트인 lmfit는 Python의 curve_fit과 API가 매우 유사합니다.
  3. Eigen 버전: diagonal().asDiagonal()은 Eigen 3.3 이상에서 안정적으로 동작합니다.
  4. 병렬화: compute_residuals와 compute_jacobian은 데이터 포인트 간 독립적이므로 #pragma omp parallel for로 쉽게 병렬화 가능합니다.

이 코드는 Eigen만 링크하면 바로 컴파일되며, scipy.optimize.curve_fit의 핵심 동작을 C++에서 재현합니다. 필요에 따라 분석적 야코비안 또는 가중치(Weighted Least Squares) 기능을 확장하면 됩니다.
