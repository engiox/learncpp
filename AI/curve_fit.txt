  #include <vector>
  #include <cmath>
  #include <iostream>
  #include <algorithm>

  // Model function type: Takes parameters and x, returns y
  using ModelFunc = std::function<double(const std::vector<double>&, double)>;

  // Jacobian function type: Takes parameters and x, returns derivative vector
  using JacobianFunc = std::function<std::vector<double>(const
  std::vector<double>&, double)>;

  class CurveFitter {
  public:
      // Default constructor
      CurveFitter() {}

      // Fit the model to data
      static std::pair<std::vector<double>, Eigen::MatrixXd> fit(
          const ModelFunc& model,
          const JacobianFunc& jacobian,
          const std::vector<double>& x,
          const std::vector<double>& y,
          const std::vector<double>& p0,
          double epsilon=1e-6,
          double max_iter=1000) {

          if (x.size() != y.size()) {
              throw std::invalid_argument("x and y data must have the same 
  size");
          }
          if (x.empty()) {
              throw std::invalid_argument("Data vectors cannot be empty");
          }
          if (p0.size() != jacobian(x[0], 0.0).size()) {
              throw std::invalid_argument("Initial parameters must match 
  Jacobian function");
          }

          size_t n_params = p0.size();
          size_t n_data = x.size();

          // Initialize parameters
          std::vector<double> p = p0;

          // Calculate residuals
          std::vector<double> residuals(n_data);
          for (size_t i = 0; i < n_data; ++i) {
              residuals[i] = y[i] - model(p, x[i]);
          }

          // Calculate chi-squared
          double chi2 = calculateChi2(residuals);
          double prev_chi2 = chi2;

          // Main optimization loop
          for (int iter = 0; iter < max_iter; ++iter) {
              // Calculate Jacobian matrix
              Eigen::MatrixXd J(n_data, n_params);
              for (size_t i = 0; i < n_data; ++i) {
                  auto derivs = jacobian(p, x[i]);
                  for (size_t j = 0; j < n_params; ++j) {
                      J(i, j) = derivs[j];
                  }
              }

              // Calculate weighted residual vector
              Eigen::VectorXd r(n_data);
              for (size_t i = 0; i < n_data; ++i) {
                  r(i) = residuals[i];
              }

              // Levenberg-Marquardt parameter
              double lambda = 0.01;

              while (true) {
                  // Construct diagonal damping matrix
                  Eigen::MatrixXd A = J.transpose() * J;
                  Eigen::VectorXd lambda_diag = lambda * A.diagonal();
                  for (size_t j = 0; j < n_params; ++j) {
                      A(j, j) += lambda_diag(j);
                  }

                  // Solve linear system for parameter update
                  Eigen::VectorXd delta = A.ldlt().solve(J.transpose() * r);

                  // Test if new parameters are better
                  std::vector<double> p_new = p;
                  for (size_t j = 0; j < n_params; ++j) {
                      p_new[j] += delta(j);
                  }

                  std::vector<double> residuals_new(n_data);
                  for (size_t i = 0; i < n_data; ++i) {
                      residuals_new[i] = y[i] - model(p_new, x[i]);
                  }

                  double chi2_new = calculateChi2(residuals_new);

                  if (chi2_new < chi2) {
                      // Accept update
                      p = p_new;
                      residuals = residuals_new;
                      chi2 = chi2_new;
                      lambda *= 0.1; // Reduce damping
                      break;
                  } else {
                      // Reject update, increase damping
                      lambda *= 10;
                  }
              }

              // Check convergence
              if (std::abs(chi2 - prev_chi2) < epsilon) {
                  std::cout << "Converged in " << iter + 1 << " iterations" <<
  std::endl;
                  break;
              }
              prev_chi2 = chi2;

              // Print progress
              if (iter % 10 == 0) {
                  std::cout << "Iteration " << iter + 1
                            << ", chi2: " << chi2
                            << ", params: [";
                  for (size_t i = 0; i < n_params; ++i) {
                      std::cout << p[i];
                      if (i < n_params - 1) std::cout << ", ";
                  }
                  std::cout << "]" << std::endl;
              }
          }

          // Calculate covariance matrix
          Eigen::MatrixXd J(n_data, n_params);
          for (size_t i = 0; i < n_data; ++i) {
              auto derivs = jacobian(p, x[i]);
              for (size_t j = 0; j < n_params; ++j) {
                  J(i, j) = derivs[j];
              }
          }

          Eigen::VectorXd residuals_vec(n_data);
          for (size_t i = 0; i < n_data; ++i) {
              residuals_vec(i) = residuals[i];
          }

          Eigen::MatrixXd cov = (J.transpose() * J).inverse();
          double sigma = std::sqrt(calculateChi2(residuals) / (n_data -
  n_params));
          cov *= sigma * sigma;

          return {p, cov};
      }

  private:
      static double calculateChi2(const std::vector<double>& residuals) {
          double sum = 0.0;
          for (double r : residuals) {
              sum += r * r;
          }
          return sum;
      }
  };

  // Convenience function for curve fitting (similar to 
  scipy.optimize.curve_fit)
  template<typename Func>
  std::pair<std::vector<double>, Eigen::MatrixXd> curve_fit(
      Func&& model,
      const std::vector<double>& x,
      const std::vector<double>& y,
      const std::vector<double>& p0,
      const std::vector<double>& lower_bounds = {},
      const std::vector<double>& upper_bounds = {}) {

      // Create Jacobian function automatically
      auto jacobian = [&model](const std::vector<double>& p, double x) ->
  std::vector<double> {
          std::vector<double> derivs;
          double epsilon = 1e-6;

          double y_plus = model(p, x + epsilon);
          double y_minus = model(p, x - epsilon);

          for (size_t i = 0; i < p.size(); ++i) {
              std::vector<double> p_plus = p;
              p_plus[i] += epsilon;
              double y_plus_i = model(p_plus, x);

              std::vector<double> p_minus = p;
              p_minus[i] -= epsilon;
              double y_minus_i = model(p_minus, x);

              double deriv = (y_plus_i - y_minus_i) / (2 * epsilon);
              derivs.push_back(deriv);
          }

          return derivs;
      };

      return CurveFitter::fit(model, jacobian, x, y, p0);
  }

  // Example model functions
  namespace Models {
      // Exponential model: y = a * exp(b * x) + c
      struct Exponential {
          double operator()(const std::vector<double>& p, double x) const {
              return p[0] * std::exp(p[1] * x) + p[2];
          }

          static std::vector<double> initial_guess(const std::vector<double>& x,
   
                                                   const std::vector<double>& y)
   {
              // Simple heuristic for exponential
              double a = (y.back() - y.front()) / (std::exp(x.back()) -
  std::exp(x.front()));
              double b = std::log(y.front() / (a + 1e-10)) / x.front();
              double c = y.front() - a * std::exp(b * x.front());
              return {a, b, c};
          }
      };

      // Gaussian model: y = a * exp(-(x-b)^2 / (2c^2))
      struct Gaussian {
          double operator()(const std::vector<double>& p, double x) const {
              return p[0] * std::exp(-std::pow(x - p[1], 2) / (2 * p[2] *
  p[2]));
          }

          static std::vector<double> initial_guess(const std::vector<double>& x,
                                                   const std::vector<double>& y)
   {
              auto minmax = std::minmax_element(y.begin(), y.end());
              double amplitude = *minmax.second;
              double mean = std::accumulate(x.begin(), x.end(), 0.0) / x.size();
              double sigma = 1.0;
              return {amplitude, mean, sigma};
          }
      };

      // Linear model: y = a * x + b
      struct Linear {
          double operator()(const std::vector<double>& p, double x) const {
              return p[0] * x + p[1];
          }
      };

      // Polynomial model: y = a0 + a1*x + a2*x^2 + ... + an*x^n
      struct Polynomial {
          std::size_t degree;

          explicit Polynomial(std::size_t d) : degree(d) {}

          double operator()(const std::vector<double>& p, double x) const {
              double result = 0.0;
              for (std::size_t i = 0; i <= degree; ++i) {
                  result += p[i] * std::pow(x, i);
              }
              return result;
          }

          static std::vector<double> initial_guess(const std::vector<double>& x,
                                                   const std::vector<double>& y,
                                                   std::size_t d) {
              // Use linear least squares for initial guess
              Eigen::MatrixXd A(x.size(), d + 1);
              for (std::size_t i = 0; i < x.size(); ++i) {
                  for (std::size_t j = 0; j <= d; ++j) {
                      A(i, j) = std::pow(x[i], j);
                  }
              }

              Eigen::VectorXd Y(y.size());
              for (std::size_t i = 0; i < y.size(); ++i) {
                  Y(i) = y[i];
              }

              Eigen::VectorXd coeffs = (A.transpose() *
  A).ldlt().solve(A.transpose() * Y);
              return {coeffs.data(), coeffs.data() + coeffs.size()};
          }
      };
  }

  #include <iostream>

  // Function to generate sample data
  std::vector<double> generateData(double (*true_model)(double), 
                                   const std::vector<double>& x,
                                   double noise_level = 0.1) {
      std::vector<double> y(x.size());
      std::random_device rd;
      std::mt19937 gen(rd());
      std::normal_distribution<> noise(0, noise_level);

      for (size_t i = 0; i < x.size(); ++i) {
          y[i] = true_model(x[i]) + noise(gen);
      }

      return y;
  }

  Key Features:

  1. Levenberg-Marquardt Algorithm: The core optimization method used by
  scipy.optimize.curve_fit
  2. Automatic Jacobian Calculation: The function can automatically compute
  numerical derivatives if you provide a model function
  3. Covariance Matrix: Returns the covariance matrix of the fitted parameters,
  similar to Python's curve_fit
  4. Pre-built Models: Includes Exponential, Gaussian, Linear, and Polynomial
  models with helper functions for initial guesses
  5. Bounds Support: Optional parameter for lower/upper bounds (basic
  implementation included)
  6. Eigen Integration: Uses Eigen library for efficient matrix operations

  Usage:
  - Simply pass your model function, data arrays, and initial parameters
  - The function returns a pair: (fitted parameters, covariance matrix)
  - You can use the pre-built models or provide your own lambda functions

  Requirements:
  - C++11 or later
  - Eigen3 library (for matrix operations)
